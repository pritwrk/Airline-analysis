SQL

VIEW Create Unified Analytical View — flights_clean
CREATE VIEW · JOIN ▼
SQL
CREATE VIEW flights_clean AS
SELECT
    f.*,
    al.AIRLINE           AS AIRLINE_NAME,
    ap_orig.CITY         AS ORIGIN_CITY,
    ap_orig.STATE        AS ORIGIN_STATE,
    ap_dest.CITY         AS DESTINATION_CITY,
    ap_dest.STATE        AS DESTINATION_STATE
FROM flights f
JOIN airlines al      ON f.AIRLINE             = al.IATA_CODE
JOIN airports ap_orig ON f.ORIGIN_AIRPORT      = ap_orig.IATA_CODE
JOIN airports ap_dest ON f.DESTINATION_AIRPORT = ap_dest.IATA_CODE;

Phase 03 EDA & KPI Queries
KPI 1 Overall Flight Performance Summary
AVG · SUM · ROUND ▼
SQL
SELECT
    COUNT(*)                                          AS total_flights,
    SUM(CANCELLED)                                    AS total_cancelled,
    ROUND(SUM(CANCELLED) * 100.0 / COUNT(*), 2)   AS cancellation_rate,
    SUM(DIVERTED)                                     AS total_diverted,
    ROUND(AVG(ARRIVAL_DELAY), 2)                    AS avg_arrival_delay,
    ROUND(AVG(DEPARTURE_DELAY), 2)                  AS avg_departure_delay

KPI 2 Airline-wise Performance Benchmarking
GROUP BY · CASE WHEN ▼
SQL
SELECT
    AIRLINE_NAME,
    COUNT(*)                                                              AS total_flights,
    ROUND(AVG(ARRIVAL_DELAY), 2)                                        AS avg_arrival_delay,
    ROUND(AVG(DEPARTURE_DELAY), 2)                                      AS avg_departure_delay,
    SUM(CANCELLED)                                                        AS total_cancelled,
    ROUND(SUM(CANCELLED) * 100.0 / COUNT(*), 2)                      AS cancellation_rate,
    ROUND(SUM(CASE WHEN ARRIVAL_DELAY <= 15 THEN 1 ELSE 0 END) * 100.0
        / COUNT(*), 2)                                                  AS on_time_rate
FROM flights_clean
GROUP BY AIRLINE_NAME
ORDER BY on_time_rate DESC;
